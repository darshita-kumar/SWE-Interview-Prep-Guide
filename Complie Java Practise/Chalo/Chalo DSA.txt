    int halfSum;
    boolean canCut;

    public boolean cutsHalf(TreeNode root){
        totalSum = sumOfTree(root);
        if(totalSum%2 != 0) return false;
        
        halfSum = totalSum/2;
        
        canCut = false;
        cutsHalfHelper(root);
        return canCut;
    }

    public int cutsHalfHelper(TreeNode root){
        if(root == null) return 0;
        int leftSum = cutsHalfHelper(root.left);
        int rightSum = cutsHalfHelper(root.right);
        int val = root.val;
        
        int total = val + leftSum + rightSum;
        
        if(total == halfSum) canCut = true;
        
        return total;
    }

    public int sumOfTree(TreeNode root){
        if(root == null) return 0;
        int leftSum = cutsHalfHelper(root.left);
        int rightSum = cutsHalfHelper(root.right);
        int val = root.val;
        int total = val + leftSum + rightSum;
        return total;
    }


}
